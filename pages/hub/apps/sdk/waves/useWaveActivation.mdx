import { Callout } from 'components'

Hook is used to [Activate User Levels](/hub/apps/guides/azuro-waves#activate-user-levels).

<Callout type="info">
  This hook is a wrapper around [activateWave](/hub/apps/toolkit/waves/activateWave) function.
</Callout>

#### Usage

```ts
import { useWaveActivation } from '@azuro-org/sdk'

const { activate } = useWaveActivation()

await activate()
```

#### Props

```ts
{
  account: Address
  waveId?: WaveId
  chainId?: ChainId
}
```

```ts
type ChainId = 
  | 100     // Gnosis
  | 137     // Polygon
  | 80002   // Polygon Amoy
  | 88888   // Chiliz
  | 88882   // Chiliz Spicy
  | 8453    // Base
  | 84532   // Base Sepolia
  
type WaveId = number | 'active'
```

#### Return Value

```ts
{
  activate: UseMutateFunction
  activateAsync: UseMutateAsyncFunction
  isPending: boolean
}
```

```ts
import { UseMutateFunction, UseMutateAsyncFunction } from '@tanstack/react-query'
```
