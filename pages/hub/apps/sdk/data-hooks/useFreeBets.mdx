import { Callout } from 'components'

Returns the [FreeBets](/contracts/free-bet) list.

<Callout type="info">
  This hook is a wrapper around [getFreeBets](/hub/apps/toolkit/utils/getFreeBets) function.
</Callout>

#### Usage

```ts
import { useFreeBets } from '@azuro-org/sdk'

const { data, isFetching } = useFreeBets({
  account: '...',
  affiliate: '...'
})
```

#### Props
```ts
{
  account: Address
  affiliate: Address
  enabled?: boolean
}
```


#### Return Value

```ts
UseQueryResult<FreeBet[]>
```

```ts
import { type UseQueryResult } from '@tanstack/react-query'

type FreeBet = {
  id: number
  contractAddress: Address
  signature: Hex
  expiresAt: number
  amount: string
  rawAmount: bigint
  minOdds: string
  rawMinOdds: bigint
  campaign: string
  chainId: ChainId
}
```
