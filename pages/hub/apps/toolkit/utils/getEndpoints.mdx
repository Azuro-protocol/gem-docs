#### Usage
```ts
import { 
  getFeedGraphqlEndpoint, 
  getBetsGraphqlEndpoint, 
  getLegacyLiveGraphqlEndpoint, 
  getSocketEndpoint, 
  getApiEndpoint 
} from '@azuro-org/toolkit'

const chainId = 100

const feedGraphqlEndpoint = getFeedGraphqlEndpoint(chainId)

const betsGraphqlEndpoint = getBetsGraphqlEndpoint(chainId)

const legacyLiveGraphqlEndpoint = getLegacyLiveGraphqlEndpoint(chainId)

const socketEndpoint = getSocketEndpoint(chainId)

const apiEndpoint = getApiEndpoint(chainId)
```

#### Props
```ts
type ChainId =

  | 88882   // Chiliz Spicy
```
